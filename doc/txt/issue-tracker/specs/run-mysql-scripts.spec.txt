	doSpecRunMysqlScripts
	
	# Purpose:
	# ------------------------------------------------------
	the purpose of this func is to run all the mysql scripts under 
   a configured mysql dir and fail on the first error 

	# Prequisites
	# ------------------------------------------------------
	mysql or mariadb installed

	# Installation
	# ------------------------------------------------------
   check the preq script

	# Interdependancies
	# ------------------------------------------------------
	- mysql installed
   - conf files with defined mysql connection params

	# Usage:
	# ------------------------------------------------------
   bash src/bash/issue-tracker/issue-tracker.sh -a run-mysql-scripts

	# Caveats:
	# ------------------------------------------------------
   - the cnf files might not be part of the package
	- none known
